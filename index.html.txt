  <script>
    // Контакты
    const CONTACTS = {
      address: "Казань, ул. Щапова 47/7, 4 этаж",
      bossName: "Настя",
      bossPhonePretty: "+7 902 289-75-28",
      bossPhoneRaw: "79022897528"
    };

    const $ = (id) => document.getElementById(id);

    const toast = (text) => {
      const el = $("toast");
      el.textContent = text;
      el.classList.add("show");
      clearTimeout(window.__t);
      window.__t = setTimeout(() => el.classList.remove("show"), 2200);
    };

    const normalizePhone = (raw) => (raw || "")
      .replace(/[^\d+]/g, "")
      .replace(/^8/, "+7");

    const buildMessage = ({name, phone, vacancy, contact, msg}) => {
      const lines = [
        `Привет! Хочу к вам в GRAFIKA (Казань).`,
        `Вакансия: ${vacancy}`,
        `Имя: ${name}`,
        `Телефон: ${phone}`,
        `Связь: ${contact}`,
        `Адрес студии: ${CONTACTS.address}`
      ];
      if (msg && msg.trim().length) lines.push(`О себе: ${msg.trim()}`);
      lines.push(`Готов(а) на короткое интервью и пробный формат.`);
      return lines.join("\n");
    };

    const openWhatsApp = (text) => {
      const url = `https://wa.me/${CONTACTS.bossPhoneRaw}?text=${encodeURIComponent(text)}`;
      window.open(url, "_blank", "noopener,noreferrer");
    };

    // Футер контакты
    $("contactsLine").textContent =
      `Руководитель: ${CONTACTS.bossName} • ${CONTACTS.bossPhonePretty} • ${CONTACTS.address}`;

    // Копирование адреса
    const copyAddress = async () => {
      try{
        await navigator.clipboard.writeText(CONTACTS.address);
        toast("Адрес скопирован.");
      }catch{
        toast("Не смог скопировать. Бывает.");
      }
    };
    $("copyAddressBtn").addEventListener("click", copyAddress);
    $("copyAddressTop").addEventListener("click", copyAddress);

    // Быстрые кнопки вакансий
    document.querySelectorAll(".applyBtn").forEach(btn => {
      btn.addEventListener("click", () => {
        $("vacancy").value = btn.dataset.vacancy || "";
        document.querySelector("#apply").scrollIntoView({behavior:"smooth"});
        toast("Вакансия выбрана. Заполни пару полей и отправляй.");
      });
    });

    // Скопировать текст отклика
    $("copyPitchBtn").addEventListener("click", async () => {
      const sample = [
        "Привет! Хочу к вам в GRAFIKA (Казань).",
        "Вакансия: (выбери)",
        "Имя: (твоё имя)",
        "Телефон: (твой номер)",
        "Коротко о себе: (опыт/график/почему ты тащишь)",
        `Адрес студии: ${CONTACTS.address}`,
        "Готов(а) на мини-интервью и пробный формат."
      ].join("\n");
      try{
        await navigator.clipboard.writeText(sample);
        toast("Скопировано. Вставляй куда надо.");
      }catch{
        toast("Не смог скопировать. Бывает.");
      }
    });

    // Отправка формы
    $("applyForm").addEventListener("submit", (e) => {
      e.preventDefault();

      const name = $("name").value.trim();
      const phone = normalizePhone($("phone").value.trim());
      const vacancy = $("vacancy").value;
      const contact = $("contact").value;
      const msg = $("msg").value;

      if (!name || !phone || !vacancy) {
        toast("Заполни имя, телефон и вакансию.");
        return;
      }

      const text = buildMessage({name, phone, vacancy, contact, msg});
      openWhatsApp(text);
      toast("Открыл WhatsApp. Отправь сообщение.");
    });

    // Tabs: day in life
    const tabBtns = document.querySelectorAll(".tabBtn");
    const panels = {
      promoter: document.querySelector("#tab-promoter"),
      call: document.querySelector("#tab-call"),
      sales: document.querySelector("#tab-sales"),
      supervisor: document.querySelector("#tab-supervisor")
    };

    tabBtns.forEach(btn => {
      btn.addEventListener("click", () => {
        tabBtns.forEach(b => b.classList.remove("active"));
        btn.classList.add("active");

        const key = btn.dataset.tab;
        Object.values(panels).forEach(p => p.classList.remove("active"));
        panels[key].classList.add("active");
      });
    });

    // Quiz
    const quizState = new Map(); // v -> 0/1
    const quizEl = document.querySelector("#quiz");
    const resEl = document.querySelector("#quizResult");
    const resetBtn = document.querySelector("#quizReset");

    const renderQuizResult = () => {
      const answered = quizState.size;
      const total = 8;
      const score = [...quizState.values()].reduce((a,b)=>a+b,0);

      if (answered === 0){
        resEl.innerHTML = `<b>Результат:</b> выбери ответы, и я скажу, насколько ты совпадаешь по стилю.`;
        return;
      }

      if (answered < total){
        resEl.innerHTML = `<b>Промежуточно:</b> ${score}/${total}. Ответь на все, чтобы было честно.`;
        return;
      }

      let text = "";
      if (score <= 3) text = "Скорее мимо. Здесь будет тяжело и раздражающе.";
      else if (score <= 6) text = "Можно пробовать. Если готов(а) учиться и держать темп, зайдёт.";
      else text = "Ты наш(а) по стилю. Откликайся, не тяни.";

      resEl.innerHTML = `<b>Результат:</b> ${score}/${total}. ${text}`;
    };

    quizEl.querySelectorAll("button[data-v]").forEach(btn => {
      btn.addEventListener("click", () => {
        const v = Number(btn.dataset.v);
        const a = Number(btn.dataset.a);

        // Активность кнопок в рамках вопроса
        const group = btn.closest(".yn");
        group.querySelectorAll("button").forEach(b => b.classList.remove("active"));
        btn.classList.add("active");

        quizState.set(v, a);
        renderQuizResult();
      });
    });

    resetBtn.addEventListener("click", () => {
      quizState.clear();
      quizEl.querySelectorAll("button[data-v]").forEach(b => b.classList.remove("active"));
      renderQuizResult();
      toast("Сброшено.");
    });

    // Slider
    const slider = document.querySelector("[data-slider]");
    if (slider){
      const track = slider.querySelector(".slides");
      const slides = Array.from(slider.querySelectorAll(".slide"));
      const dotsWrap = slider.querySelector("[data-dots]");
      const prev = slider.querySelector("[data-prev]");
      const next = slider.querySelector("[data-next]");

      let index = 0;
      let timer = null;

      const setIndex = (i) => {
        index = (i + slides.length) % slides.length;
        track.style.transform = `translateX(-${index * 100}%)`;

        dotsWrap.querySelectorAll(".dotBtn").forEach((d, di) => {
          d.classList.toggle("active", di === index);
        });
      };

      // dots
      slides.forEach((_, i) => {
        const b = document.createElement("button");
        b.type = "button";
        b.className = "dotBtn" + (i===0 ? " active" : "");
        b.addEventListener("click", () => setIndex(i));
        dotsWrap.appendChild(b);
      });

      prev.addEventListener("click", () => setIndex(index - 1));
      next.addEventListener("click", () => setIndex(index + 1));

      // autoplay
      const start = () => {
        stop();
        timer = setInterval(() => setIndex(index + 1), 5200);
      };
      const stop = () => {
        if (timer) clearInterval(timer);
        timer = null;
      };
      slider.addEventListener("mouseenter", stop);
      slider.addEventListener("mouseleave", start);
      start();

      // swipe
      let x0 = null;
      slider.addEventListener("touchstart", (e) => { x0 = e.touches[0].clientX; }, {passive:true});
      slider.addEventListener("touchend", (e) => {
        if (x0 === null) return;
        const x1 = e.changedTouches[0].clientX;
        const dx = x1 - x0;
        if (Math.abs(dx) > 40){
          if (dx > 0) setIndex(index - 1);
          else setIndex(index + 1);
        }
        x0 = null;
      }, {passive:true});
    }

    // Reveal on scroll
    const revealEls = document.querySelectorAll(".reveal");
    const io = new IntersectionObserver((entries) => {
      entries.forEach((e) => {
        if (e.isIntersecting){
          e.target.classList.add("in");
          io.unobserve(e.target);
        }
      });
    }, {threshold: 0.12});
    revealEls.forEach(el => io.observe(el));
